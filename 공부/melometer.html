Chat

문서명 : 커플ID

필드

  • chatType - 채팅유형 ( text, image 등 )
  • contents - 채팅 텍스트/사진URL 등 내용
  • date(nanoseconds) - 입력시간(나노초)
  • date(second) - 입력시간(초)
  • messageId - 각 매세지 별 고유 ID
  • userID - 매새시 송신자 UUID

Users

문서명 : UUID

필드

  • accesssLevel - 가입단계 (가입중 / 로그인 / 탈퇴 등 )
  • createdAt - 가입일자 시간
  • fcmToken - fcm토큰 정보
  • inviteCode - 가입시 기입 Number ( 12334234 )
  • phoneNumber - 전화번호
  • stateMessage - 상태메세지
  • uid - UUID

CREATE TABLE Alarm (
AlarmID UUID PRIMARY KEY,
Date DATE NOT NULL,
Text TEXT NOT NULL,
Type VARCHAR(255) NOT NULL
);

CREATE TABLE Chat (
ChatID UUID PRIMARY KEY,
ChatType VARCHAR(50) NOT NULL,
Contents TEXT NOT NULL,
DateNano BIGINT NOT NULL,
DateSec BIGINT NOT NULL,
MessageID UUID NOT NULL,
UserID UUID NOT NULL
);

CREATE TABLE Couples (
CoupleID UUID PRIMARY KEY,
AnniDate1 DATE,
AnniDate2 DATE,
AnniName1 VARCHAR(255),
AnniName2 VARCHAR(255)
);

CREATE TABLE Answer (
AnswerID UUID PRIMARY KEY,
CoupleID UUID REFERENCES Couples(CoupleID),
Date DATE NOT NULL,
AnswerText TEXT NOT NULL,
UserID UUID NOT NULL,
UserName VARCHAR(255) NOT NULL
);

CREATE TABLE Locations (
UserID UUID PRIMARY KEY,
Latitude FLOAT NOT NULL,
Longitude FLOAT NOT NULL
);

CREATE TABLE QuestionList (
QuestionID INT PRIMARY KEY,
QuestionText TEXT NOT NULL
);

CREATE TABLE Users (
UserID UUID PRIMARY KEY,
AccessLevel VARCHAR(50) NOT NULL,
CreatedAt TIMESTAMP NOT NULL,
FCMToken VARCHAR(255),
InviteCode INT,
PhoneNumber VARCHAR(20),
StateMessage VARCHAR(255)
);

CREATE TABLE Users (
UUID VARCHAR(50) NOT NULL,
CoupleID VARCHAR(50) NOT NULL,
userID INT NOT NULL,
AccessLevel INT NOT NULL,
CreateDate DATE NOT NULL,
FCMTokken VARCHAR(50),
InviteCode INT,
PhoneNumber VARCHAR(50) NOT NULL,
StateMessage VARCHAR(50),
BirthDate DATE NOT NULL,
PRIMARY KEY (UUID)
);

CREATE TABLE Couples (
CoupleID VARCHAR(50) NOT NULL,
UUID1 VARCHAR(50) NOT NULL,
UUID2 VARCHAR(50) NOT NULL,
PRIMARY KEY (CoupleID),
FOREIGN KEY (UUID1) REFERENCES Users(UUID),
FOREIGN KEY (UUID2) REFERENCES Users(UUID)
);

CREATE TABLE Location (
UUID VARCHAR(50) NOT NULL,
CoupleID VARCHAR(50) NOT NULL,
Latitude FLOAT,
Longitude FLOAT,
PRIMARY KEY (UUID),
FOREIGN KEY (CoupleID) REFERENCES Couples(CoupleID)
);

CREATE TABLE Alarm (
UUID VARCHAR(50) NOT NULL,
CoupleID VARCHAR(50) NOT NULL,
Date DATE,
Text VARCHAR(50),
Type VARCHAR(50) NOT NULL,
PRIMARY KEY (UUID),
FOREIGN KEY (CoupleID) REFERENCES Couples(CoupleID)
);

CREATE TABLE Chat (
ChatID INT NOT NULL,
UUID VARCHAR(50) NOT NULL,
CoupleID VARCHAR(50) NOT NULL,
ChatType INT NOT NULL,
Contents VARCHAR(500) NOT NULL,
ChatDate DATE NOT NULL,
PRIMARY KEY (ChatID),
FOREIGN KEY (UUID) REFERENCES Users(UUID),
FOREIGN KEY (CoupleID) REFERENCES Couples(CoupleID)
);

CREATE TABLE QuestionList (
QID INT NOT NULL,
QText VARCHAR(500) NOT NULL,
PRIMARY KEY (QID)
);

CREATE TABLE Answer (
AID INT NOT NULL,
QID INT NOT NULL,
UUID VARCHAR(50) NOT NULL,
CoupleID VARCHAR(50) NOT NULL,
ADate DATE NOT NULL,
AText VARCHAR(500) NOT NULL,
PRIMARY KEY (AID),
FOREIGN KEY (QID) REFERENCES QuestionList(QID),
FOREIGN KEY (UUID) REFERENCES Users(UUID),
FOREIGN KEY (CoupleID) REFERENCES Couples(CoupleID)
);

CREATE TABLE Anni (
UUID VARCHAR(50) NOT NULL,
CoupleID VARCHAR(50) NOT NULL,
anniDate DATE,
anniName VARCHAR(50),
PRIMARY KEY (UUID),
FOREIGN KEY (CoupleID) REFERENCES Couples(CoupleID)
);
