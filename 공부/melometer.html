CREATE TABLE Users (
UUID VARCHAR(50) NOT NULL,
CoupleID VARCHAR(50) NOT NULL,
userID INT NOT NULL,
AccessLevel INT NOT NULL,
CreateDate DATE NOT NULL,
FCMTokken VARCHAR(50),
InviteCode INT,
PhoneNumber VARCHAR(50) NOT NULL,
StateMessage VARCHAR(50),
BirthDate DATE NOT NULL,
PRIMARY KEY (UUID)
);

CREATE TABLE Couples (
CoupleID VARCHAR(50) NOT NULL,
UUID1 VARCHAR(50) NOT NULL,
UUID2 VARCHAR(50) NOT NULL,
PRIMARY KEY (CoupleID),
FOREIGN KEY (UUID1) REFERENCES Users(UUID),
FOREIGN KEY (UUID2) REFERENCES Users(UUID)
);

CREATE TABLE Location (
UUID VARCHAR(50) NOT NULL,
CoupleID VARCHAR(50) NOT NULL,
Latitude FLOAT,
Longitude FLOAT,
PRIMARY KEY (UUID),
FOREIGN KEY (CoupleID) REFERENCES Couples(CoupleID)
);

CREATE TABLE Alarm (
UUID VARCHAR(50) NOT NULL,
CoupleID VARCHAR(50) NOT NULL,
Date DATE,
Text VARCHAR(50),
Type VARCHAR(50) NOT NULL,
PRIMARY KEY (UUID),
FOREIGN KEY (CoupleID) REFERENCES Couples(CoupleID)
);

CREATE TABLE Chat (
ChatID INT NOT NULL,
UUID VARCHAR(50) NOT NULL,
CoupleID VARCHAR(50) NOT NULL,
ChatType INT NOT NULL,
Contents VARCHAR(500) NOT NULL,
ChatDate DATE NOT NULL,
PRIMARY KEY (ChatID),
FOREIGN KEY (UUID) REFERENCES Users(UUID),
FOREIGN KEY (CoupleID) REFERENCES Couples(CoupleID)
);

CREATE TABLE QuestionList (
QID INT NOT NULL,
QText VARCHAR(500) NOT NULL,
PRIMARY KEY (QID)
);

AID INT NOT NULL,

ADate DATE NOT NULL,
AText VARCHAR(500) NOT NULL,
PRIMARY KEY (AID),
